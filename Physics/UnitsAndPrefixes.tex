	\begin{center}
%
		\textbf{\underline{Example Conversion Problem}} \\
		\underline{Convert $6.3$ ft$/$s to m$/$s}
%
	\end{center}
%
	\[
%
		6.3 \text{ ft}/\text{ s} = \frac{6.3 \text{ ft}}{1 \text{ s}}  = \frac{6.3*1\text{ ft}}{1\text{ s}}
%
	\]
%
	\begin{center}
%
		($1$ m $=$ $3.281 * 1$ ft, so $0.305 * 1$ m $=$ $1$ ft, which can be substituted) \\
%
	\end{center}
%
	\[
%
		\frac{6.3 * 1 \text{ ft}}{1 \text{ s}} = \frac{6.3*0.305*1 \text{ m}}{1 \text{ s}} = \frac{1.922 \text{ m}}{1 \text{ s}} = 1.922\text{ m}/\text{s}
%
	\]

	\begin{center}
%
		\textbf{\underline{Example Conversion Problem}} \\
		\underline{Convert $65.2$ mm to nm}
%
	\end{center}
%
	\[
%		
		\begin{split}
%
			65.2 \text{ mm} &= 65.2 * 1 \text{ mm } \\
					&= 65.2 * 10^{-3} * 1 \text{ m }\\
%
		\end{split}
%
	\]
%
	\begin{center}
%
		($1$ nm $=$ $10^{-6} * 1$ m, so $10^6 * 1$ nm $=$ $1$ m, which can be substituted) \\
%
	\end{center}
%
	\[
%			
		\begin{split}
%	
			65.2 * 10^{-3} * 1 \text{ m } &= 65.2 * 10^{-3} * 10^6 * 1 \text{ nm } \\
						      &= 65.2 * 10^3 \text{ nm } \\
						      &= 65200 \text{ nm }
%
		\end{split}
%
	\]
%
	\begin{center}
%
		Note, this is \emph{one} way to solve this problem, notably without explicit use of conversion factors.
%
	\end{center}
